title: "getLassoElementTypeCounts"
description: "获取套索元素类型数量统计。"
---

```ts
static getLassoElementTypeCounts(): Promise<APIResponse<LassoElementTypeNum>>;
```

**返回**

- `APIResponse<LassoElementTypeNum>`：`result` 见 [`LassoElementTypeNum`](/api-reference/supernote-plugin/types/lasso-trail-type-num)

## 示例

```ts wrap
import { PluginCommAPI } from 'rtn-supernote-plugin-core';

/**
 * 获取套索元素类型数量统计的示例。
 */
export async function exampleGetLassoElementTypeCounts() {
  const res = await PluginCommAPI.getLassoElementTypeCounts();
  if (!res.success) {
    throw new Error(res.error?.message ?? 'getLassoElementTypeCounts 调用失败');
  }
  return res.result;
}
```
